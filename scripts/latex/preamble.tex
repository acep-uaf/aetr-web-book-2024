% Define custom header and footer
\newcommand{\headerfooter}{
  \fancyhf{}
  \fancyhead[L]{{\hspace*{-\marginparsep}\hspace*{-\marginparwidth}\hspace*{-1in}\makebox[\dimexpr\paperwidth+2in\relax]{\color{uafblue}\rule{\dimexpr\paperwidth+2in\relax}{40pt}}}

  \fancyhead[R]{\raisebox{30pt}{\color{white} \textbf{\fontsize{14}{16}\selectfont Alaska Electricity Trends Report, 2024}}}
  \fancyfoot[C]{\raisebox{60pt}{\vspace{0pt}\pagemark}}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
  % Header footer spacing
  \setlength{\headheight}{70pt}
  \setlength{\footskip}{20pt}
  \setlength{\topmargin}{-85pt}
}

% Set header and footer for chapter title pages
\fancypagestyle{chaptertitle}{

% Set header and footer for for TOC and LOF pages
\fancypagestyle{toclof}{

% Apply custom page style to TOC and LOF

% Chapter title formatting
\makeatletter
\def\@makechapterhead#1{%
  \thispagestyle{chaptertitle} % Apply custom page style to chapter title pages
  \vspace*{50pt} % Space before the chapter title
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \huge\bfseries \color{uafblue} \@chapapp\space \thechapter % Chapter number
      \par\nobreak
      \vskip 0pt % Space between chapter number and title
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak % Chapter title
}}
\makeatother
