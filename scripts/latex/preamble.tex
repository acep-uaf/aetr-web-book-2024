\makeatletter
\def\@makechapterhead#1{%
  \vspace*{50pt} % Space before the chapter title
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \huge\bfseries \color{uafblue} \@chapapp\space \thechapter % Chapter number
      \par\nobreak
      \vskip 0pt % Space between chapter number and title
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak % Chapter title
    \vskip 0pt % Space after the chapter title
  }}
\def\@makeschapterhead#1{%
  \vspace*{0pt} % Space before the chapter title (unnumbered)
  {\parindent \z@ \raggedright \normalfont
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak % Chapter title (unnumbered)
    \vskip 0pt % Space after the chapter title (unnumbered)
  }}
