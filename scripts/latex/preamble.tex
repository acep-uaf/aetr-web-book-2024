% Define custom header and footer
\newcommand{\headerfooter}{
  \fancyhf{}
  \fancyhead[C]{
    \makebox[0pt][c]{\hspace*{0cm}\color{uafblue}\rule{\dimexpr\textwidth+2cm\relax}{40pt}}
    \makebox[0pt][c]{\hspace*{-0.25cm}\color{uafyellow}\rule{\dimexpr\textwidth+2cm\relax}{5pt}}
  }
  \fancyhead[R]{\raisebox{17pt}{\color{white} \textbf{\fontsize{14}{16}\selectfont Alaska Electricity Trends Report, 2024}}}
  \fancyfoot[R]{\raisebox{0pt}{\color{uafblue} \textbf{\thepage}}}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

% Set header and footer for chapter title pages
\fancypagestyle{chaptertitle}{
  \headerfooter}
% Set header and footer for for TOC and LOF pages
\fancypagestyle{toclof}{
  \headerfooter}
% Apply custom page style to TOC and LOF
\addtocontents{toc}{\protect\thispagestyle{toclof}}
\addtocontents{lof}{\protect\thispagestyle{toclof}}
\addtocontents{lot}{\protect\thispagestyle{toclof}}

% Chapter title formatting
\makeatletter
\def\@makechapterhead#1{%
  \thispagestyle{chaptertitle} % Apply custom page style to chapter title pages
  \vspace*{-30pt} % Space before the chapter title
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \huge\bfseries \color{uafblue} \@chapapp\space \thechapter % Chapter number
      \par\nobreak
      \vskip 0pt % Space between chapter number and title
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak % Chapter title
}}
\makeatother
